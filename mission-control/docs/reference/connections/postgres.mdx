```yaml
apiVersion: mission-control.flanksource.com/v1
kind: Connection
metadata:
  name: payments-database
  namespace: default
spec:
  postgres:
    database:
      value: payments
    host:
      value: postgres.host.com
    insecureTLS: true
    username:
      valueFrom:
        secretKeyRef:
          name: payments-database-credentials
          key: POSTGRES_USER
    password:
      valueFrom:
        secretKeyRef:
          name: payments-database-credentials
          key: POSTGRES_PASSWORD
```
